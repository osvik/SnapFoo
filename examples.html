Animate a Sequence

This example demonstrates animating and staggering a group of circle elements.

var snapfoo = snapFoo("#sequence-animation");
    var theValues = {
        keyframes: [
            { values: {transform: "t200,0"}, duration: 1000, easing: mina.backout },
            { values: {transform: "t200,0, s.75,.75"}, duration: 500, easing: mina.bounce },
            { values: {transform: "t400,0, s.75,.75"}, duration: 1000, easing: mina.backout }
        ],
        stagger: 1000
    };
    snapfoo.animate("circle", theValues);

This example demonstrates looping an animation that rotates and element while changing its points. Additionally, it utilizes the keyframeDelay value.

var theValues = {
    keyframes: [
        { values: {transform: "r90,250,75", points: "286.99,111.99 250.426,75 213.01,111.99 249.405,74.467 213.01,38.01 249.405,75 286.99,38.01 286.99,73.958"}, duration: 1000 },
        { values: {transform: "r180,250,75", points: "286.99,111.99 250.426,111.99 213.01,111.99 249.405,74.467 213.01,38.01 249.405,75 286.99,38.01 286.99,73.958"}, duration: 1000 },
        { values: {transform: "r270,250,75", points: "286.99,111.99 250.426,111.99 213.01,111.99 213.01,74.467 213.01,38.01 249.405,75 286.99,38.01 286.99,73.958"}, duration: 1000 },
        { values: {transform: "r360,250,75", points: "286.99,111.99 250.426,111.99 213.01,111.99 213.01,74.467 213.01,38.01 249.405,38.01 286.99,38.01 286.99,73.958"}, duration: 1000 },

Animate Using Callbacks

This example demonstrates the use of different callback types to affect an element and chain animations.

var snapfoo = snapFoo("#callback-animation");
var theValues = {
    keyframes: [
        { values: {transform: "t200,0"}, duration: 1000, easing: mina.backout },
        { values: {transform: "t200,0, s.75,.75"}, duration: 500, easing: mina.bounce },
        { values: {transform: "t400,0, s.75,.75"}, duration: 1000, easing: mina.backout, options: { keyframeCallback: function() {
            snapfoo.set("#"+$('#callback-animation circle[class^="sf"]').last().attr("id"),{fill: "#ff0000"}); }} },
        { values: {transform: "t0,0"}, duration: 1000, easing: mina.bounce }
    ],
    animationCallback: function() {
        snapfoo.animate("#"+$('#callback-animation circle[class^="sf"]:last').next("circle").attr("id"),theValues);
        $('#callback-animation circle[class^="sf"]:last').attr('class',
            function(i, c){
                return c.replace(/(^|\s)sf\S+/g, '');
        }
    ); }
};
snapfoo.animate("#circle-one", theValues);

Animate Along a Path

This example demonstrates animating along a path without any additional options.

var snapfoopath = snapFoo("#animate-path");
snapfoopath.animatePath("#the-ghost", "#the-path", 2500);

Animate a Loop Along a Path

This example demonstrates looping along a path while reversing the direction of the path.

var snapfooPathLoop = snapFoo("#animate-path-loop");
var theOptions = {
    loop: true,
    reversePathDirection: true
};
snapfooPathLoop.animatePath("#the-ghost2", "#the-path2", 2500, theOptions);

Animate a Loop Along a Path with Reverse Options

This example demonstrates looping along a path where the element is reversed at the end of the path and separate reverse options have been set.

var snapfooPathLoop = snapFoo("#animate-path-loop-reverse");
var theOptions = {
    loop: true,
    reverseAtEnd: true,
    reverseOptions: {
        duration: 1000,
        easing: mina.easeout,
        callback: function() {console.log("Reverse Ended!");}
    }
};
snapfooPathLoop.animatePath("#ghost", "#path", 2500, theOptions);

Animate a Loop Along a Path with Reverse Options

This example demonstrates looping along a path where the element is reversed at the end of the path and separate reverse options have been set.

var snapfooPathGroup = snapFoo("#animate-path-loop-group");
var theOptions = {
    loop: true,
    stagger: 1000
};
snapfooPathGroup.animatePath("circle", "#group-path", 2500, theOptions);
